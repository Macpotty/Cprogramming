#include<iostream>
#include<string>
using namespace std;
const char *findA(char const *source, char const *obj){
	for (unsigned int i=0; i < strlen(obj); i++){
		for (unsigned int j = 0; j < strlen(source); j++){
			if (*(obj + i) == *(source + j)){
				return source;
			}
		}
	}
	return NULL;
}
const char *findC(char const *source, char const *obj){
	for (unsigned int i = 0; i < strlen(obj); i++){
		for (unsigned int j = 0; j < strlen(source); j++){
			if (*(obj + i) == *(source + j)){
				if (i == strlen(obj))
					return obj;
				else {
					for (i, j; i < strlen(obj) || j < strlen(source); i++, j++){
						if (*(obj + i) == *(source + j))
							continue;
						else return NULL;
					}
					if (i + 1 == strlen(obj)&&j+1>=strlen(source)){
						return obj;
					}
					else return NULL;
				}
			}
			else return NULL;
		}
	}
	return NULL;
}
